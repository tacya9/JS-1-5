### 5.5.4 Сортировать в порядке по убыванию let arr = [5, 2, 1, -10, 8]; alert( arr ); // 8, 5, 2, 1, -10 А еще отсортировать так: числа сортируются по убыванию, а булевое значение true должно стать в самом начале, а false - в самом конце: [5, 2, 1, -10, 8, 12, true, false, true, false, 20, -3] отсортируется так: [true, true, 20, 12, 8, 5, 2, 1, -3, -10, false, false]
